# zmalloc()жәҗз Ғ

void *zmalloc(size_t size) {
    void *ptr = malloc(size+PREFIX_SIZE);
 
    if (!ptr) zmalloc_oom_handler(size);
#ifdef HAVE_MALLOC_SIZE
    update_zmalloc_stat_alloc(zmalloc_size(ptr));
    return ptr;
#else
    *((size_t*)ptr) = size;
    update_zmalloc_stat_alloc(size+PREFIX_SIZE);
    return (char*)ptr+PREFIX_SIZE;
#endif
}

еҸӮж•°sizeжҳҜжҲ‘д»¬ йңҖиҰҒеҲҶй…Қзҡ„еҶ…еӯҳеӨ§е°ҸгҖӮе®һйҷ…дёҠжҲ‘д»¬и°ғз”Ёmalloc е®һйҷ…еҲҶй…Қзҡ„еӨ§е°ҸжҳҜsize+PREFIX_SIZEгҖӮPREFIX_SIZEжҳҜдёҖдёӘжқЎд»¶зј–иҜ‘зҡ„е®ҸпјҢдёҚеҗҢзҡ„е№іеҸ°жңүдёҚеҗҢзҡ„з»“жһңпјҢеңЁLinuxдёӯе…¶еҖјжҳҜsizeof(size_t)пјҢжүҖд»ҘжҲ‘д»¬еӨҡеҲҶй…ҚдәҶдёҖдёӘеӯ—й•ҝ(8дёӘеӯ—иҠӮ)зҡ„з©әй—ҙпјҲеҗҺйқўд»Јз ҒеҸҜд»ҘзңӢеҲ°еӨҡеҲҶй…Қ8дёӘеӯ—иҠӮзҡ„зӣ®зҡ„жҳҜз”ЁдәҺеӮЁеӯҳsizeзҡ„еҖјпјүгҖӮ еҰӮжһңptrжҢҮй’ҲдёәNULLпјҲеҶ…еӯҳеҲҶй…ҚеӨұиҙҘпјүпјҢи°ғз”Ёzmalloc_oom_handlerпјҲsizeпјүгҖӮиҜҘеҮҪж•°е®һйҷ…дёҠжҳҜдёҖдёӘеҮҪж•°жҢҮй’ҲжҢҮеҗ‘еҮҪж•°zmalloc_default_oomпјҢе…¶дё»иҰҒеҠҹиғҪе°ұжҳҜжү“еҚ°й”ҷиҜҜдҝЎжҒҜе№¶з»ҲжӯўзЁӢеәҸгҖӮ // oomжҳҜout of memoryпјҲеҶ…еӯҳдёҚи¶іпјүзҡ„ж„ҸжҖқ

static void zmalloc_default_oom(size_t size) {
    fprintf(stderr, "zmalloc: Out of memory trying to allocate %zu bytes\n",
        size);
    fflush(stderr);
    abort();
}

жҺҘдёӢжқҘжҳҜе®Ҹзҡ„жқЎд»¶зј–иҜ‘пјҢжҲ‘д»¬иҒҡз„ҰеңЁ#elseзҡ„йғЁеҲҶгҖӮ

*((size_t*)ptr) = size;
update_zmalloc_stat_alloc(size+PREFIX_SIZE);

з¬¬дёҖиЎҢе°ұжҳҜеңЁе·ІеҲҶй…Қз©әй—ҙзҡ„з¬¬дёҖдёӘеӯ—й•ҝпјҲеүҚ8дёӘеӯ—иҠӮпјүеӨ„еӯҳеӮЁ йңҖиҰҒеҲҶй…Қзҡ„еӯ—иҠӮеӨ§е°ҸпјҲsizeпјүгҖӮ

з¬¬дәҢиЎҢи°ғз”ЁдәҶupdate_zmalloc_stat_alloc()гҖҗе®ҸеҮҪж•°гҖ‘пјҢе®ғзҡ„еҠҹиғҪжҳҜжӣҙж–°е…ЁеұҖеҸҳйҮҸused_memoryпјҲе·ІеҲҶй…ҚеҶ…еӯҳзҡ„еӨ§е°Ҹпјүзҡ„еҖјпјҲжәҗз Ғи§ЈиҜ»и§ҒдёӢдёҖиҠӮпјүгҖӮ

з¬¬дёүиЎҢиҝ”еӣһзҡ„пјҲchar *пјүptr+PREFIX_SIZEгҖӮе°ұжҳҜе°Ҷе·ІеҲҶй…ҚеҶ…еӯҳзҡ„иө·е§Ӣең°еқҖеҗ‘еҸіеҒҸз§»PREFIX_SIZE * sizeof(char)зҡ„й•ҝеәҰпјҲеҚі8дёӘеӯ—иҠӮпјүпјҢжӯӨж—¶еҫ—еҲ°зҡ„ж–°жҢҮй’ҲжҢҮеҗ‘зҡ„еҶ…еӯҳз©әй—ҙзҡ„еӨ§е°Ҹе°ұзӯүдәҺsizeдәҶгҖӮ

жҺҘдёӢжқҘпјҢеҲҶжһҗдёҖдёӢupdate_zmalloc_stat_allocзҡ„жәҗз Ғ

# update_zmalloc_stat_allocжәҗз Ғ

#define update_zmalloc_stat_alloc(__n) do { \
    size_t _n = (__n); \
    if (_n&(sizeof(long)-1)) _n += sizeof(long)-(_n&(sizeof(long)-1)); \
    if (zmalloc_thread_safe) { \
        update_zmalloc_stat_add(_n); \
    } else { \
        used_memory += _n; \
    } \
} while(0)

иҝҷдёӘе®ҸеҮҪж•°жңҖеӨ–еңҲжңүдёҖдёӘdo{...}while(0)еҫӘзҺҜзңӢдјјжҜ«ж— ж„Ҹд№үпјҢе®һйҷ…дёҠеӨ§жңүж·ұж„ҸгҖӮиҝҷйғЁеҲҶеҶ…е®№дёҚжҳҜжң¬ж–Үи®Ёи®әзҡ„йҮҚзӮ№пјҢиҝҷйҮҢдёҚеҶҚиөҳиҝ°гҖӮе…·дҪ“иҜ·зңӢзҪ‘дёҠзҡ„иҝҷзҜҮж–Үз«  http://www.spongeliu.com/415.htmlгҖӮ еӣ дёә sizeof(long) = 8 гҖҗ64дҪҚзі»з»ҹдёӯгҖ‘пјҢжүҖд»ҘдёҠйқўзҡ„з¬¬дёҖдёӘifиҜӯеҸҘпјҢеҸҜд»Ҙзӯүд»·дәҺд»ҘдёӢд»Јз Ғпјҡ

 if(_n&7) _n += 8 - (_n&7);

иҝҷж®өд»Јз Ғе°ұжҳҜеҲӨж–ӯеҲҶй…Қзҡ„еҶ…еӯҳз©әй—ҙзҡ„еӨ§е°ҸжҳҜдёҚжҳҜ8зҡ„еҖҚж•°гҖӮеҰӮжһңеҶ…еӯҳеӨ§е°ҸдёҚжҳҜ8зҡ„еҖҚж•°пјҢе°ұеҠ дёҠзӣёеә”зҡ„еҒҸз§»йҮҸдҪҝд№ӢеҸҳжҲҗ8зҡ„еҖҚж•°гҖӮ_n&7 еңЁеҠҹиғҪдёҠзӯүд»·дәҺ _n%8пјҢдёҚиҝҮдҪҚж“ҚдҪңзҡ„ж•ҲзҺҮжҳҫз„¶жӣҙй«ҳгҖӮ malloc()жң¬иә«иғҪеӨҹдҝқиҜҒжүҖеҲҶй…Қзҡ„еҶ…еӯҳжҳҜ8еӯ—иҠӮеҜ№йҪҗзҡ„пјҡеҰӮжһңдҪ иҰҒеҲҶй…Қзҡ„еҶ…еӯҳдёҚжҳҜ8зҡ„еҖҚж•°пјҢйӮЈд№Ҳmallocе°ұдјҡеӨҡеҲҶй…ҚдёҖзӮ№пјҢжқҘеҮ‘жҲҗ8зҡ„еҖҚж•°гҖӮжүҖд»Ҙupdate_zmalloc_stat_allocеҮҪж•°пјҲжҲ–иҖ…иҜҙzmalloc()зӣёеҜ№malloc()иҖҢиЁҖпјүзңҹжӯЈиҰҒе®һзҺ°зҡ„еҠҹиғҪ е№¶дёҚжҳҜиҝӣиЎҢ8еӯ—иҠӮеҜ№йҪҗпјҲmallocе·Із»ҸдҝқиҜҒдәҶпјүпјҢе®ғзҡ„зңҹжӯЈзӣ®зҡ„жҳҜдҪҝеҸҳйҮҸused_memoryзІҫзЎ®зҡ„з»ҙжҠӨе®һйҷ…е·ІеҲҶй…ҚеҶ…еӯҳзҡ„еӨ§е°ҸгҖӮ
з¬¬2дёӘifзҡ„жқЎд»¶жҳҜдёҖдёӘж•ҙеһӢеҸҳйҮҸzmalloc_thread_safeгҖӮйЎҫеҗҚжҖқд№үпјҢе®ғзҡ„еҖјиЎЁзӨәж“ҚдҪңжҳҜеҗҰжҳҜзәҝзЁӢе®үе…Ёзҡ„пјҢеҰӮжһңдёҚжҳҜзәҝзЁӢе®үе…Ёзҡ„пјҲelseпјүпјҢе°ұз»ҷеҸҳйҮҸused_memoryеҠ дёҠnгҖӮused_memoryжҳҜzmalloc.cж–Үд»¶дёӯе®ҡд№үзҡ„е…ЁеұҖйқҷжҖҒеҸҳйҮҸпјҢиЎЁзӨәе·ІеҲҶй…ҚеҶ…еӯҳзҡ„еӨ§е°ҸгҖӮеҰӮжһңжҳҜеҶ…еӯҳе®үе…Ёзҡ„е°ұдҪҝз”Ёupdate_zmalloc_stat_addжқҘз»ҷused_memoryеҠ дёҠnгҖӮ update_zmalloc_stat_addд№ҹжҳҜдёҖдёӘе®ҸеҮҪж•°пјҲRedisж•ҲзҺҮд№Ӣй«ҳпјҢйҖҹеәҰд№Ӣеҝ«пјҢиҝҷдәӣе®ҸеҮҪж•°еҸҜи°“еҠҹдёҚеҸҜжІЎпјүгҖӮе®ғд№ҹжҳҜдёҖдёӘжқЎд»¶зј–иҜ‘зҡ„е®ҸпјҢдҫқжҚ®дёҚеҗҢзҡ„е®ҸжңүдёҚеҗҢзҡ„е®ҡд№үпјҢиҝҷйҮҢжҲ‘д»¬жқҘзңӢдёҖдёӢ#elseеҗҺйқўзҡ„е®ҡд№үзҡ„жәҗз ҒгҖҗzmalloc.cжңүеӨҡеӨ„жқЎд»¶зј–иҜ‘зҡ„е®ҸпјҢдёәдәҶжҠҠзІҫеҠӣйғҪйӣҶдёӯеңЁеҶ…еӯҳз®ЎзҗҶзҡ„е®һзҺ°з®—жі•дёҠпјҢиҝҷйҮҢжҲ‘еҸӘе…іжіЁLinuxе№іеҸ°дёӢдҪҝз”Ёglibcзҡ„mallocзҡ„жғ…еҶөгҖ‘гҖӮ

#define update_zmalloc_stat_add(__n) do { \
    pthread_mutex_lock(&used_memory_mutex); \
    used_memory += (__n); \
    pthread_mutex_unlock(&used_memory_mutex); \
} while(0)

pthread_mutex_lock()е’Ңpthread_mutex_unlock()дҪҝз”Ёдә’ж–Ҙй”ҒпјҲmutexпјүжқҘе®һзҺ°зәҝзЁӢеҗҢжӯҘпјҢеүҚиҖ…иЎЁзӨәеҠ й”ҒпјҢеҗҺиҖ…иЎЁзӨәи§Јй”ҒпјҢе®ғд»¬жҳҜPOSIXе®ҡд№үзҡ„зәҝзЁӢеҗҢжӯҘеҮҪж•°гҖӮеҪ“еҠ й”Ғд»ҘеҗҺе®ғеҗҺйқўзҡ„д»Јз ҒеңЁеӨҡзәҝзЁӢеҗҢж—¶жү§иЎҢиҝҷж®өд»Јз Ғзҡ„ж—¶еҖҷе°ұеҸӘдјҡжү§иЎҢдёҖж¬ЎпјҢд№ҹе°ұжҳҜе®һзҺ°дәҶзәҝзЁӢе®үе…ЁгҖӮ

# zfree()жәҗз Ғ

void zfree(void *ptr) {
#ifndef HAVE_MALLOC_SIZE
    void *realptr;
    size_t oldsize;
#endif
 
    if (ptr == NULL) return;
#ifdef HAVE_MALLOC_SIZE
    update_zmalloc_stat_free(zmalloc_size(ptr));
    free(ptr);
#else
    realptr = (char*)ptr-PREFIX_SIZE;
    oldsize = *((size_t*)realptr);
    update_zmalloc_stat_free(oldsize+PREFIX_SIZE);
    free(realptr);
#endif
}

йҮҚзӮ№е…іжіЁ#elseеҗҺйқўзҡ„д»Јз Ғ

realptr = (char *)ptr - PREFIX_SIZE;

иЎЁзӨәзҡ„жҳҜptrжҢҮй’Ҳеҗ‘еүҚеҒҸз§»8дёӘеӯ—иҠӮзҡ„й•ҝеәҰпјҢеҚіеӣһйҖҖеҲ°жңҖеҲқmallocиҝ”еӣһзҡ„ең°еқҖпјҢиҝҷйҮҢз§°дёәrealptrгҖӮз„¶еҗҺ

oldsize = *((size_t*)realptr);

# update_zmalloc_freeжәҗз Ғ

#define update_zmalloc_stat_free(__n) do { \
    size_t _n = (__n); \
    if (_n&(sizeof(long)-1)) _n += sizeof(long)-(_n&(sizeof(long)-1)); \
    if (zmalloc_thread_safe) { \
        update_zmalloc_stat_sub(_n); \
    } else { \
        used_memory -= _n; \
    } \
} while(0)

е…¶дёӯзҡ„еҮҪж•°update_zmalloc_subдёҺzmalloc()дёӯзҡ„update_zmalloc_addзӣёеҜ№еә”пјҢдҪҶеҠҹиғҪзӣёеҸҚпјҢжҸҗдҫӣзәҝзЁӢе®үе…Ёең°used_memoryеҮҸжі•ж“ҚдҪңгҖӮ

#define update_zmalloc_stat_sub(__n) do { \
    pthread_mutex_lock(&used_memory_mutex); \
    used_memory -= (__n); \
    pthread_mutex_unlock(&used_memory_mutex); \
} while(0)

void *zcalloc(size_t size) {
    void *ptr = calloc(1, size+PREFIX_SIZE);
 
    if (!ptr) zmalloc_oom_handler(size);
#ifdef HAVE_MALLOC_SIZE
    update_zmalloc_stat_alloc(zmalloc_size(ptr));
    return ptr;
#else
    *((size_t*)ptr) = size;
    update_zmalloc_stat_alloc(size+PREFIX_SIZE);
    return (char*)ptr+PREFIX_SIZE;
#endif
}

    zcalloc()е’Ңzmalloc()е…·жңүзӣёеҗҢзҡ„зј–зЁӢжҺҘеҸЈпјҢе®һзҺ°еҠҹиғҪеҹәжң¬зӣёеҗҢпјҢе”ҜдёҖдёҚеҗҢд№ӢеӨ„жҳҜzcalloc()дјҡеҒҡеҲқе§ӢеҢ–е·ҘдҪңпјҢиҖҢzmalloc()дёҚдјҡгҖӮ

# zrealloc()жәҗз Ғ

void *zrealloc(void *ptr, size_t size) {
#ifndef HAVE_MALLOC_SIZE
    void *realptr;
#endif
    size_t oldsize;
    void *newptr;
 
    if (ptr == NULL) return zmalloc(size);
#ifdef HAVE_MALLOC_SIZE
    oldsize = zmalloc_size(ptr);
    newptr = realloc(ptr,size);
    if (!newptr) zmalloc_oom_handler(size);
 
    update_zmalloc_stat_free(oldsize);
    update_zmalloc_stat_alloc(zmalloc_size(newptr));
    return newptr;
#else
    realptr = (char*)ptr-PREFIX_SIZE;
    oldsize = *((size_t*)realptr);
    newptr = realloc(realptr,size+PREFIX_SIZE);
    if (!newptr) zmalloc_oom_handler(size);
 
    *((size_t*)newptr) = size;
    update_zmalloc_stat_free(oldsize);
    update_zmalloc_stat_alloc(size);
    return (char*)newptr+PREFIX_SIZE;
#endif
}
