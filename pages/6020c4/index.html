# дёҖгҖҒSynchronizedдҪҝз”ЁеңәжҷҜ

SynchronizedжҳҜдёҖдёӘеҗҢжӯҘе…ій”®еӯ—пјҢеңЁжҹҗдәӣеӨҡзәҝзЁӢеңәжҷҜдёӢпјҢеҰӮжһңдёҚиҝӣиЎҢеҗҢжӯҘдјҡеҜјиҮҙж•°жҚ®дёҚе®үе…ЁпјҢиҖҢSynchronizedе…ій”®еӯ—е°ұжҳҜз”ЁдәҺд»Јз ҒеҗҢжӯҘгҖӮд»Җд№Ҳжғ…еҶөдёӢдјҡж•°жҚ®дёҚе®үе…Ёе‘ўпјҢиҰҒж»Ўи¶ідёӨдёӘжқЎд»¶пјҡдёҖжҳҜж•°жҚ®е…ұдә«пјҲдёҙз•Ңиө„жәҗпјүпјҢдәҢжҳҜеӨҡзәҝзЁӢеҗҢж—¶и®ҝй—®е№¶ж”№еҸҳиҜҘж•°жҚ®гҖӮ

дҫӢеҰӮпјҡ

public class AccountingSync implements Runnable{
    //е…ұдә«иө„жәҗ(дёҙз•Ңиө„жәҗ)
    static int i=0;
 
    /**
     * synchronized дҝ®йҘ°е®һдҫӢж–№жі•
     */
    public synchronized void increase(){
        i++;
    }
    @Override
    public void run() {
        for(int j=0;j<1000000;j++){
            increase();
        }
    }
    public static void main(String[] args) throws InterruptedException {
        AccountingSync instance=new AccountingSync();
        Thread t1=new Thread(instance);
        Thread t2=new Thread(instance);
        t1.start();
        t2.start();
        t1.join();
        t2.join();
        System.out.println(i);
    }
}

иҜҘж®өзЁӢеәҸзҡ„иҫ“еҮәдёәпјҡ2000000

дҪҶжҳҜеҰӮжһңincreaseзҡ„synchronizedиў«еҲ йҷӨпјҢйӮЈд№ҲеҫҲеҸҜиғҪиҫ“еҮәз»“жһңе°ұдјҡе°ҸдәҺ2000000пјҢиҝҷжҳҜеӣ дёәеӨҡдёӘзәҝзЁӢеҗҢж—¶и®ҝй—®дёҙз•Ңиө„жәҗiпјҢеҰӮжһңдёҖдёӘзәҝзЁӢAеҜ№i=88зҡ„иҮӘеўһеҲ°89жІЎжңүиў«BзәҝзЁӢиҜ»еҸ–еҲ°пјҢзәҝзЁӢBи®Өдёәiд»Қз„¶жҳҜ88пјҢйӮЈд№ҲзәҝзЁӢBеҜ№iзҡ„иҮӘеўһз»“жһңиҝҳжҳҜ89пјҢйӮЈд№ҲиҝҷйҮҢе°ұдјҡеҮәзҺ°й—®йўҳгҖӮ

Synchronizedй”Ғзҡ„3з§ҚдҪҝз”ЁеҪўејҸпјҲдҪҝз”ЁеңәжҷҜпјүпјҡ

  • Synchronizedдҝ®йҘ°жҷ®йҖҡеҗҢжӯҘж–№жі•пјҡй”ҒеҜ№иұЎеҪ“еүҚе®һдҫӢеҜ№иұЎпјӣ
  • Synchronizedдҝ®йҘ°йқҷжҖҒеҗҢжӯҘж–№жі•пјҡй”ҒеҜ№иұЎжҳҜеҪ“еүҚзҡ„зұ»ClassеҜ№иұЎпјӣ
  • Synchronizedдҝ®йҘ°еҗҢжӯҘд»Јз Ғеқ—пјҡй”ҒеҜ№иұЎжҳҜSynchronizedеҗҺйқўжӢ¬еҸ·йҮҢй…ҚзҪ®зҡ„еҜ№иұЎпјҢиҝҷдёӘеҜ№иұЎеҸҜд»ҘжҳҜжҹҗдёӘеҜ№иұЎпјҲxlockпјүпјҢд№ҹеҸҜд»ҘжҳҜжҹҗдёӘзұ»пјҲXlock.classпјүпјӣ

жіЁж„Ҹпјҡ

  • дҪҝз”Ёsynchronizedдҝ®йҘ°йқһйқҷжҖҒж–№жі•жҲ–иҖ…дҪҝз”Ёsynchronizedдҝ®йҘ°д»Јз Ғеқ—ж—¶еҲ¶е®ҡзҡ„дёәе®һдҫӢеҜ№иұЎж—¶пјҢеҗҢдёҖдёӘзұ»зҡ„дёҚеҗҢеҜ№иұЎжӢҘжңүиҮӘе·ұзҡ„й”ҒпјҢеӣ жӯӨдёҚдјҡзӣёдә’йҳ»еЎһгҖӮ
  • дҪҝз”Ёsynchronizedдҝ®йҘ°зұ»е’ҢеҜ№иұЎж—¶пјҢз”ұдәҺзұ»еҜ№иұЎе’Ңе®һдҫӢеҜ№иұЎеҲҶеҲ«жӢҘжңүиҮӘе·ұзҡ„зӣ‘и§ҶеҷЁй”ҒпјҢеӣ жӯӨдёҚдјҡзӣёдә’йҳ»еЎһгҖӮ
  • дҪҝз”ЁдҪҝз”Ёsynchronizedдҝ®йҘ°е®һдҫӢеҜ№иұЎж—¶пјҢеҰӮжһңдёҖдёӘзәҝзЁӢжӯЈеңЁи®ҝй—®е®һдҫӢеҜ№иұЎзҡ„дёҖдёӘsynchronizedж–№жі•ж—¶пјҢе…¶е®ғзәҝзЁӢдёҚд»…дёҚиғҪи®ҝй—®иҜҘsynchronizedж–№жі•пјҢиҜҘеҜ№иұЎзҡ„е…¶е®ғsynchronizedж–№жі•д№ҹдёҚиғҪи®ҝй—®пјҢеӣ дёәдёҖдёӘеҜ№иұЎеҸӘжңүдёҖдёӘзӣ‘и§ҶеҷЁй”ҒеҜ№иұЎпјҢдҪҶжҳҜе…¶е®ғзәҝзЁӢеҸҜд»Ҙи®ҝй—®иҜҘеҜ№иұЎзҡ„йқһsynchronizedж–№жі•гҖӮ
  • зәҝзЁӢAи®ҝй—®е®һдҫӢеҜ№иұЎзҡ„йқһstatic synchronizedж–№жі•ж—¶пјҢзәҝзЁӢBд№ҹеҸҜд»ҘеҗҢж—¶и®ҝй—®е®һдҫӢеҜ№иұЎзҡ„static synchronizedж–№жі•пјҢеӣ дёәеүҚиҖ…иҺ·еҸ–зҡ„жҳҜе®һдҫӢеҜ№иұЎзҡ„зӣ‘и§ҶеҷЁй”ҒпјҢиҖҢеҗҺиҖ…иҺ·еҸ–зҡ„жҳҜзұ»еҜ№иұЎзҡ„зӣ‘и§ҶеҷЁй”ҒпјҢдёӨиҖ…дёҚеӯҳеңЁдә’ж–Ҙе…ізі»гҖӮ

# дәҢгҖҒSynchronizedе®һзҺ°еҺҹзҗҶ

1гҖҒJavaеҜ№иұЎеӨҙ

йҰ–е…ҲпјҢжҲ‘д»¬иҰҒзҹҘйҒ“еҜ№иұЎеңЁеҶ…еӯҳдёӯзҡ„еёғеұҖпјҡ

е·ІзҹҘеҜ№иұЎжҳҜеӯҳж”ҫеңЁе ҶеҶ…еӯҳдёӯзҡ„пјҢеҜ№иұЎеӨ§иҮҙеҸҜд»ҘеҲҶдёәдёүдёӘйғЁеҲҶпјҢеҲҶеҲ«жҳҜеҜ№иұЎеӨҙгҖҒе®һдҫӢеҸҳйҮҸе’ҢеЎ«е……еӯ—иҠӮгҖӮ

  • еҜ№иұЎеӨҙзҡ„zhuyaoжҳҜз”ұMarkWordе’ҢKlass Point(зұ»еһӢжҢҮй’Ҳ)з»„жҲҗпјҢе…¶дёӯKlass PointжҳҜжҳҜеҜ№иұЎжҢҮеҗ‘е®ғзҡ„зұ»е…ғж•°жҚ®зҡ„жҢҮй’ҲпјҢиҷҡжӢҹжңәйҖҡиҝҮиҝҷдёӘжҢҮй’ҲжқҘзЎ®е®ҡиҝҷдёӘеҜ№иұЎжҳҜе“ӘдёӘзұ»зҡ„е®һдҫӢпјҢMark Wordз”ЁдәҺеӯҳеӮЁеҜ№иұЎиҮӘиә«зҡ„иҝҗиЎҢж—¶ж•°жҚ®гҖӮеҰӮжһңеҜ№иұЎжҳҜж•°з»„еҜ№иұЎпјҢйӮЈд№ҲеҜ№иұЎеӨҙеҚ з”Ё3дёӘеӯ—е®ҪпјҲWordпјүпјҢеҰӮжһңеҜ№иұЎжҳҜйқһж•°з»„еҜ№иұЎпјҢйӮЈд№ҲеҜ№иұЎеӨҙеҚ з”Ё2дёӘеӯ—е®ҪгҖӮпјҲ1word = 2 Byte = 16 bitпјү
  • е®һдҫӢеҸҳйҮҸеӯҳеӮЁзҡ„жҳҜеҜ№иұЎзҡ„еұһжҖ§дҝЎжҒҜпјҢеҢ…жӢ¬зҲ¶зұ»зҡ„еұһжҖ§дҝЎжҒҜпјҢжҢүз…§4еӯ—иҠӮеҜ№йҪҗ
  • еЎ«е……еӯ—з¬ҰпјҢеӣ дёәиҷҡжӢҹжңәиҰҒжұӮеҜ№иұЎеӯ—иҠӮеҝ…йЎ»жҳҜ8еӯ—иҠӮзҡ„ж•ҙж•°еҖҚпјҢеЎ«е……еӯ—з¬Ұе°ұжҳҜз”ЁдәҺеҮ‘йҪҗиҝҷдёӘж•ҙж•°еҖҚзҡ„

йҖҡиҝҮз¬¬дёҖйғЁеҲҶеҸҜд»ҘзҹҘйҒ“пјҢSynchronizedдёҚи®әжҳҜдҝ®йҘ°ж–№жі•иҝҳжҳҜд»Јз Ғеқ—пјҢйғҪжҳҜйҖҡиҝҮжҢҒжңүдҝ®йҘ°еҜ№иұЎзҡ„й”ҒжқҘе®һзҺ°еҗҢжӯҘпјҢйӮЈд№ҲSynchronizedй”ҒеҜ№иұЎжҳҜеӯҳеңЁе“ӘйҮҢзҡ„е‘ўпјҹзӯ”жЎҲжҳҜеӯҳеңЁй”ҒеҜ№иұЎзҡ„еҜ№иұЎеӨҙзҡ„MarkWordдёӯгҖӮйӮЈд№ҲMarkWordеңЁеҜ№иұЎеӨҙдёӯеҲ°еә•й•ҝд»Җд№Ҳж ·пјҢд№ҹе°ұжҳҜе®ғеҲ°еә•еӯҳеӮЁдәҶд»Җд№Ҳе‘ўпјҹ

еңЁ32дҪҚзҡ„иҷҡжӢҹжңәдёӯпјҡ

еңЁ64дҪҚзҡ„иҷҡжӢҹжңәдёӯпјҡ

дёҠеӣҫдёӯзҡ„еҒҸеҗ‘й”Ғе’ҢиҪ»йҮҸзә§й”ҒйғҪжҳҜеңЁjava6д»ҘеҗҺеҜ№й”ҒжңәеҲ¶иҝӣиЎҢдјҳеҢ–ж—¶еј•иҝӣзҡ„пјҢдёӢж–Үзҡ„й”ҒеҚҮзә§йғЁеҲҶдјҡе…·дҪ“и®Іи§ЈпјҢSynchronizedе…ій”®еӯ—еҜ№еә”зҡ„жҳҜйҮҚйҮҸзә§й”ҒпјҢжҺҘдёӢжқҘеҜ№йҮҚйҮҸзә§й”ҒеңЁHotspot JVMдёӯзҡ„е®һзҺ°й”Ғи®Іи§ЈгҖӮ

2гҖҒSynchronizedеңЁJVMдёӯзҡ„е®һзҺ°еҺҹзҗҶ

йҮҚйҮҸзә§й”ҒеҜ№еә”зҡ„й”Ғж Үеҝ—дҪҚжҳҜ10пјҢеӯҳеӮЁдәҶжҢҮеҗ‘йҮҚйҮҸзә§зӣ‘и§ҶеҷЁй”Ғзҡ„жҢҮй’ҲпјҢеңЁHotspotдёӯпјҢеҜ№иұЎзҡ„зӣ‘и§ҶеҷЁпјҲmonitorпјүй”ҒеҜ№иұЎз”ұObjectMonitorеҜ№иұЎе®һзҺ°пјҲC++пјүпјҢе…¶и·ҹеҗҢжӯҘзӣёе…ізҡ„ж•°жҚ®з»“жһ„еҰӮдёӢпјҡ

ObjectMonitor() {
    _count        = 0; //з”ЁжқҘи®°еҪ•иҜҘеҜ№иұЎиў«зәҝзЁӢиҺ·еҸ–й”Ғзҡ„ж¬Ўж•°
    _waiters      = 0;
    _recursions   = 0; //й”Ғзҡ„йҮҚе…Ҙж¬Ўж•°
    _owner        = NULL; //жҢҮеҗ‘жҢҒжңүObjectMonitorеҜ№иұЎзҡ„зәҝзЁӢ 
    _WaitSet      = NULL; //еӨ„дәҺwaitзҠ¶жҖҒзҡ„зәҝзЁӢпјҢдјҡиў«еҠ е…ҘеҲ°_WaitSet
    _WaitSetLock  = 0 ;
    _EntryList    = NULL ; //еӨ„дәҺзӯүеҫ…й”ҒblockзҠ¶жҖҒзҡ„зәҝзЁӢпјҢдјҡиў«еҠ е…ҘеҲ°иҜҘеҲ—иЎЁ
  }

е…үзңӢиҝҷдәӣж•°жҚ®з»“жһ„еҜ№зӣ‘и§ҶеҷЁй”Ғзҡ„е·ҘдҪңжңәеҲ¶иҝҳжҳҜдёҖеӨҙйӣҫж°ҙпјҢйӮЈд№ҲжҲ‘д»¬йҰ–е…ҲзңӢдёҖдёӢзәҝзЁӢеңЁиҺ·еҸ–й”Ғзҡ„еҮ дёӘзҠ¶жҖҒзҡ„иҪ¬жҚўпјҡ

зәҝзЁӢзҡ„з”ҹе‘Ҫе‘ЁжңҹеӯҳеңЁ5дёӘзҠ¶жҖҒпјҢstartгҖҒrunningгҖҒwaitingгҖҒblockingе’Ңdead

еҜ№дәҺдёҖдёӘsynchronizedдҝ®йҘ°зҡ„ж–№жі•(д»Јз Ғеқ—)жқҘиҜҙпјҡ

  • еҪ“еӨҡдёӘзәҝзЁӢеҗҢж—¶и®ҝй—®иҜҘж–№жі•пјҢйӮЈд№ҲиҝҷдәӣзәҝзЁӢдјҡе…Ҳиў«ж”ҫиҝӣ_EntryListйҳҹеҲ—пјҢжӯӨж—¶зәҝзЁӢеӨ„дәҺblockingзҠ¶жҖҒ
  • еҪ“дёҖдёӘзәҝзЁӢиҺ·еҸ–еҲ°дәҶе®һдҫӢеҜ№иұЎзҡ„зӣ‘и§ҶеҷЁпјҲmonitorпјүй”ҒпјҢйӮЈд№Ҳе°ұеҸҜд»Ҙиҝӣе…ҘrunningзҠ¶жҖҒпјҢжү§иЎҢж–№жі•пјҢжӯӨж—¶пјҢObjectMonitorеҜ№иұЎзҡ„_ownerжҢҮеҗ‘еҪ“еүҚзәҝзЁӢпјҢ_countеҠ 1иЎЁзӨәеҪ“еүҚеҜ№иұЎй”Ғиў«дёҖдёӘзәҝзЁӢиҺ·еҸ–
  • еҪ“runningзҠ¶жҖҒзҡ„зәҝзЁӢи°ғз”Ёwait()ж–№жі•пјҢйӮЈд№ҲеҪ“еүҚзәҝзЁӢйҮҠж”ҫmonitorеҜ№иұЎпјҢиҝӣе…ҘwaitingзҠ¶жҖҒпјҢObjectMonitorеҜ№иұЎзҡ„_ownerеҸҳдёәnullпјҢ_countеҮҸ1пјҢеҗҢж—¶зәҝзЁӢиҝӣе…Ҙ_WaitSetйҳҹеҲ—пјҢзӣҙеҲ°жңүзәҝзЁӢи°ғз”Ёnotify()ж–№жі•е”ӨйҶ’иҜҘзәҝзЁӢпјҢеҲҷиҜҘзәҝзЁӢйҮҚж–°иҺ·еҸ–monitorеҜ№иұЎиҝӣе…Ҙ_OwnerеҢә
  • еҰӮжһңеҪ“еүҚзәҝзЁӢжү§иЎҢе®ҢжҜ•пјҢйӮЈд№Ҳд№ҹйҮҠж”ҫmonitorеҜ№иұЎпјҢиҝӣе…ҘwaitingзҠ¶жҖҒпјҢObjectMonitorеҜ№иұЎзҡ„_ownerеҸҳдёәnullпјҢ_countеҮҸ1

йӮЈд№ҲSynchronizedдҝ®йҘ°зҡ„д»Јз Ғеқ—/ж–№жі•еҰӮдҪ•иҺ·еҸ–monitorеҜ№иұЎзҡ„е‘ўпјҹ

еңЁJVMи§„иҢғйҮҢеҸҜд»ҘзңӢеҲ°пјҢдёҚз®ЎжҳҜж–№жі•еҗҢжӯҘиҝҳжҳҜд»Јз Ғеқ—еҗҢжӯҘйғҪжҳҜеҹәдәҺиҝӣе…Ҙе’ҢйҖҖеҮәmonitorеҜ№иұЎжқҘе®һзҺ°пјҢз„¶иҖҢдәҢиҖ…еңЁе…·дҪ“е®һзҺ°дёҠеҸҲеӯҳеңЁеҫҲеӨ§зҡ„еҢәеҲ«гҖӮйҖҡиҝҮjavapеҜ№classеӯ—иҠӮз Ғж–Үд»¶еҸҚзј–иҜ‘еҸҜд»Ҙеҫ—еҲ°еҸҚзј–иҜ‘еҗҺзҡ„д»Јз ҒгҖӮ

пјҲ1пјүSynchronizedдҝ®йҘ°д»Јз Ғеқ—пјҡ

Synchronizedд»Јз Ғеқ—еҗҢжӯҘеңЁйңҖиҰҒеҗҢжӯҘзҡ„д»Јз Ғеқ—ејҖе§Ӣзҡ„дҪҚзҪ®жҸ’е…ҘmonitorentryжҢҮд»ӨпјҢеңЁеҗҢжӯҘз»“жқҹзҡ„дҪҚзҪ®жҲ–иҖ…ејӮеёёеҮәзҺ°зҡ„дҪҚзҪ®жҸ’е…ҘmonitorexitжҢҮд»ӨпјӣJVMиҰҒдҝқиҜҒmonitorentryе’ҢmonitorexitйғҪжҳҜжҲҗеҜ№еҮәзҺ°зҡ„пјҢд»»дҪ•еҜ№иұЎйғҪжңүдёҖдёӘmonitorдёҺд№ӢеҜ№еә”пјҢеҪ“иҝҷдёӘеҜ№иұЎзҡ„monitorиў«жҢҒжңүд»ҘеҗҺпјҢе®ғе°ҶеӨ„дәҺй”Ғе®ҡзҠ¶жҖҒгҖӮ

дҫӢеҰӮпјҢеҗҢжӯҘд»Јз Ғеқ—еҰӮдёӢпјҡ

public class SyncCodeBlock {
   public int i;
   public void syncTask(){
       synchronized (this){
           i++;
       }
   }
}

пјҲ2пјүSynchronizedдҝ®йҘ°ж–№жі•пјҡ

Synchronizedж–№жі•еҗҢжӯҘдёҚеҶҚжҳҜйҖҡиҝҮжҸ’е…Ҙmonitorentryе’ҢmonitorexitжҢҮд»Өе®һзҺ°пјҢиҖҢжҳҜз”ұж–№жі•и°ғз”ЁжҢҮд»ӨжқҘиҜ»еҸ–иҝҗиЎҢж—¶еёёйҮҸжұ дёӯзҡ„ACC_SYNCHRONIZEDж Үеҝ—йҡҗејҸе®һзҺ°зҡ„пјҢеҰӮжһңж–№жі•иЎЁз»“жһ„пјҲmethod_info Structureпјүдёӯзҡ„ACC_SYNCHRONIZEDж Үеҝ—иў«и®ҫзҪ®пјҢйӮЈд№ҲзәҝзЁӢеңЁжү§иЎҢж–№жі•еүҚдјҡе…ҲеҺ»иҺ·еҸ–еҜ№иұЎзҡ„monitorеҜ№иұЎпјҢеҰӮжһңиҺ·еҸ–жҲҗеҠҹеҲҷжү§иЎҢж–№жі•д»Јз ҒпјҢжү§иЎҢе®ҢжҜ•еҗҺйҮҠж”ҫmonitorеҜ№иұЎпјҢеҰӮжһңmonitorеҜ№иұЎе·Із»Ҹиў«е…¶е®ғзәҝзЁӢиҺ·еҸ–пјҢйӮЈд№ҲеҪ“еүҚзәҝзЁӢиў«йҳ»еЎһгҖӮ

# дёүгҖҒй”Ғзҡ„дјҳеҢ–

1гҖҒй”ҒеҚҮзә§ й”Ғзҡ„4дёӯзҠ¶жҖҒпјҡж— й”ҒзҠ¶жҖҒгҖҒеҒҸеҗ‘й”ҒзҠ¶жҖҒгҖҒиҪ»йҮҸзә§й”ҒзҠ¶жҖҒгҖҒйҮҚйҮҸзә§й”ҒзҠ¶жҖҒпјҲзә§еҲ«д»ҺдҪҺеҲ°й«ҳпјү

пјҲ1пјүеҒҸеҗ‘й”Ғпјҡ

дёәд»Җд№ҲиҰҒеј•е…ҘеҒҸеҗ‘й”Ғпјҹ

еӣ дёәз»ҸиҝҮHotSpotзҡ„дҪңиҖ…еӨ§йҮҸзҡ„з ”з©¶еҸ‘зҺ°пјҢеӨ§еӨҡж•°ж—¶еҖҷжҳҜдёҚеӯҳеңЁй”Ғз«һдәүзҡ„пјҢеёёеёёжҳҜдёҖдёӘзәҝзЁӢеӨҡж¬ЎиҺ·еҫ—еҗҢдёҖдёӘй”ҒпјҢеӣ жӯӨеҰӮжһңжҜҸж¬ЎйғҪиҰҒз«һдәүй”ҒдјҡеўһеӨ§еҫҲеӨҡжІЎжңүеҝ…иҰҒд»ҳеҮәзҡ„д»Јд»·пјҢдёәдәҶйҷҚдҪҺиҺ·еҸ–й”Ғзҡ„д»Јд»·пјҢжүҚеј•е…Ҙзҡ„еҒҸеҗ‘й”ҒгҖӮ

еҒҸеҗ‘й”Ғзҡ„еҚҮзә§

еҪ“зәҝзЁӢ1и®ҝй—®д»Јз Ғеқ—е№¶иҺ·еҸ–й”ҒеҜ№иұЎж—¶пјҢдјҡеңЁjavaеҜ№иұЎеӨҙе’Ңж Ҳеё§дёӯи®°еҪ•еҒҸеҗ‘зҡ„й”Ғзҡ„threadIDпјҢеӣ дёәеҒҸеҗ‘й”ҒдёҚдјҡдё»еҠЁйҮҠж”ҫй”ҒпјҢеӣ жӯӨд»ҘеҗҺзәҝзЁӢ1еҶҚж¬ЎиҺ·еҸ–й”Ғзҡ„ж—¶еҖҷпјҢйңҖиҰҒжҜ”иҫғеҪ“еүҚзәҝзЁӢзҡ„threadIDе’ҢJavaеҜ№иұЎеӨҙдёӯзҡ„threadIDжҳҜеҗҰдёҖиҮҙпјҢеҰӮжһңдёҖиҮҙпјҲиҝҳжҳҜзәҝзЁӢ1иҺ·еҸ–й”ҒеҜ№иұЎпјүпјҢеҲҷж— йңҖдҪҝз”ЁCASжқҘеҠ й”ҒгҖҒи§Јй”ҒпјӣеҰӮжһңдёҚдёҖиҮҙпјҲе…¶д»–зәҝзЁӢпјҢеҰӮзәҝзЁӢ2иҰҒз«һдәүй”ҒеҜ№иұЎпјҢиҖҢеҒҸеҗ‘й”ҒдёҚдјҡдё»еҠЁйҮҠж”ҫеӣ жӯӨиҝҳжҳҜеӯҳеӮЁзҡ„зәҝзЁӢ1зҡ„threadIDпјүпјҢйӮЈд№ҲйңҖиҰҒжҹҘзңӢJavaеҜ№иұЎеӨҙдёӯи®°еҪ•зҡ„зәҝзЁӢ1жҳҜеҗҰеӯҳжҙ»пјҢеҰӮжһңжІЎжңүеӯҳжҙ»пјҢйӮЈд№Ҳй”ҒеҜ№иұЎиў«йҮҚзҪ®дёәж— й”ҒзҠ¶жҖҒпјҢе…¶е®ғзәҝзЁӢпјҲзәҝзЁӢ2пјүеҸҜд»Ҙз«һдәүе°Ҷе…¶и®ҫзҪ®дёәеҒҸеҗ‘й”ҒпјӣеҰӮжһңеӯҳжҙ»пјҢйӮЈд№Ҳз«ӢеҲ»жҹҘжүҫиҜҘзәҝзЁӢпјҲзәҝзЁӢ1пјүзҡ„ж Ҳеё§дҝЎжҒҜпјҢеҰӮжһңиҝҳжҳҜйңҖиҰҒз»§з»ӯжҢҒжңүиҝҷдёӘй”ҒеҜ№иұЎпјҢйӮЈд№ҲжҡӮеҒңеҪ“еүҚзәҝзЁӢ1пјҢж’Өй”ҖеҒҸеҗ‘й”ҒпјҢеҚҮзә§дёәиҪ»йҮҸзә§й”ҒпјҢеҰӮжһңзәҝзЁӢ1 дёҚеҶҚдҪҝз”ЁиҜҘй”ҒеҜ№иұЎпјҢйӮЈд№Ҳе°Ҷй”ҒеҜ№иұЎзҠ¶жҖҒи®ҫдёәж— й”ҒзҠ¶жҖҒпјҢйҮҚж–°еҒҸеҗ‘ж–°зҡ„зәҝзЁӢгҖӮ

еҒҸеҗ‘й”Ғзҡ„еҸ–ж¶Ҳпјҡ

еҒҸеҗ‘й”ҒжҳҜй»ҳи®ӨејҖеҗҜзҡ„пјҢиҖҢдё”ејҖе§Ӣж—¶й—ҙдёҖиҲ¬жҳҜжҜ”еә”з”ЁзЁӢеәҸеҗҜеҠЁж…ўеҮ з§’пјҢеҰӮжһңдёҚжғіжңүиҝҷдёӘе»¶иҝҹпјҢйӮЈд№ҲеҸҜд»ҘдҪҝз”Ё-XX:BiasedLockingStartUpDelay=0пјӣ

еҰӮжһңдёҚжғіиҰҒеҒҸеҗ‘й”ҒпјҢйӮЈд№ҲеҸҜд»ҘйҖҡиҝҮ-XX:-UseBiasedLocking = falseжқҘи®ҫзҪ®пјӣ

пјҲ2пјүиҪ»йҮҸзә§й”Ғ

дёәд»Җд№ҲиҰҒеј•е…ҘиҪ»йҮҸзә§й”Ғпјҹ

иҪ»йҮҸзә§й”ҒиҖғиҷ‘зҡ„жҳҜз«һдәүй”ҒеҜ№иұЎзҡ„зәҝзЁӢдёҚеӨҡпјҢиҖҢдё”зәҝзЁӢжҢҒжңүй”Ғзҡ„ж—¶й—ҙд№ҹдёҚй•ҝзҡ„жғ…жҷҜгҖӮеӣ дёәйҳ»еЎһзәҝзЁӢйңҖиҰҒCPUд»Һз”ЁжҲ·жҖҒиҪ¬еҲ°еҶ…ж ёжҖҒпјҢд»Јд»·иҫғеӨ§пјҢеҰӮжһңеҲҡеҲҡйҳ»еЎһдёҚд№…иҝҷдёӘй”Ғе°ұиў«йҮҠж”ҫдәҶпјҢйӮЈиҝҷдёӘд»Јд»·е°ұжңүзӮ№еҫ—дёҚеҒҝеӨұдәҶпјҢеӣ жӯӨиҝҷдёӘж—¶еҖҷе°ұе№Іи„ҶдёҚйҳ»еЎһиҝҷдёӘзәҝзЁӢпјҢи®©е®ғиҮӘж—Ӣиҝҷзӯүеҫ…й”ҒйҮҠж”ҫгҖӮ

иҪ»йҮҸзә§й”Ғд»Җд№Ҳж—¶еҖҷеҚҮзә§дёәйҮҚйҮҸзә§й”Ғпјҹ

зәҝзЁӢ1иҺ·еҸ–иҪ»йҮҸзә§й”Ғж—¶дјҡе…ҲжҠҠй”ҒеҜ№иұЎзҡ„еҜ№иұЎеӨҙMarkWordеӨҚеҲ¶дёҖд»ҪеҲ°зәҝзЁӢ1зҡ„ж Ҳеё§дёӯеҲӣе»әзҡ„з”ЁдәҺеӯҳеӮЁй”Ғи®°еҪ•зҡ„з©әй—ҙпјҲз§°дёәDisplacedMarkWordпјүпјҢз„¶еҗҺдҪҝз”ЁCASжҠҠеҜ№иұЎеӨҙдёӯзҡ„еҶ…е®№жӣҝжҚўдёәзәҝзЁӢ1еӯҳеӮЁзҡ„й”Ғи®°еҪ•пјҲDisplacedMarkWordпјүзҡ„ең°еқҖпјӣ

*жіЁж„ҸпјҡдёәдәҶйҒҝе…Қж— з”Ёзҡ„иҮӘж—ӢпјҢиҪ»йҮҸзә§й”ҒдёҖж—ҰиҶЁиғҖдёәйҮҚйҮҸзә§й”Ғе°ұдёҚдјҡеҶҚйҷҚзә§дёәиҪ»йҮҸзә§й”ҒдәҶпјӣеҒҸеҗ‘й”ҒеҚҮзә§дёәиҪ»йҮҸзә§й”Ғд№ҹдёҚиғҪеҶҚйҷҚзә§дёәеҒҸеҗ‘й”ҒгҖӮдёҖеҸҘиҜқе°ұжҳҜй”ҒеҸҜд»ҘеҚҮзә§дёҚеҸҜд»ҘйҷҚзә§пјҢдҪҶжҳҜеҒҸеҗ‘й”ҒзҠ¶жҖҒеҸҜд»Ҙиў«йҮҚзҪ®дёәж— й”ҒзҠ¶жҖҒгҖӮ
