# и§Јзӯ”

йҷҗжөҒжҳҜдҝқжҠӨй«ҳе№¶еҸ‘зі»з»ҹзҡ„дёүжҠҠеҲ©еҷЁд№ӢдёҖпјҢеҸҰеӨ–дёӨдёӘжҳҜзј“еӯҳе’ҢйҷҚзә§гҖӮйҷҗжөҒеңЁеҫҲеӨҡеңәжҷҜдёӯз”ЁжқҘйҷҗеҲ¶е№¶еҸ‘е’ҢиҜ·жұӮйҮҸпјҢжҜ”еҰӮиҜҙз§’жқҖжҠўиҙӯпјҢдҝқжҠӨиҮӘиә«зі»з»ҹе’ҢдёӢжёёзі»з»ҹдёҚиў«е·ЁеһӢжөҒйҮҸеҶІеһ®зӯүгҖӮ

йҷҗжөҒзҡ„зӣ®зҡ„жҳҜйҖҡиҝҮеҜ№е№¶еҸ‘и®ҝй—®/иҜ·жұӮиҝӣиЎҢйҷҗйҖҹжҲ–иҖ…дёҖдёӘж—¶й—ҙзӘ—еҸЈеҶ…зҡ„зҡ„иҜ·жұӮиҝӣиЎҢйҷҗйҖҹжқҘдҝқжҠӨзі»з»ҹпјҢдёҖж—ҰиҫҫеҲ°йҷҗеҲ¶йҖҹзҺҮеҲҷеҸҜд»ҘжӢ’з»қжңҚеҠЎжҲ–иҝӣиЎҢжөҒйҮҸж•ҙеҪўгҖӮ

еёёз”Ёзҡ„йҷҗжөҒж–№ејҸе’ҢеңәжҷҜжңүпјҡйҷҗеҲ¶жҖ»е№¶еҸ‘ж•°пјҲжҜ”еҰӮж•°жҚ®еә“иҝһжҺҘжұ гҖҒзәҝзЁӢжұ пјүгҖҒйҷҗеҲ¶зһ¬ж—¶е№¶еҸ‘ж•°пјҲеҰӮnginxзҡ„limitconnжЁЎеқ—пјҢз”ЁжқҘйҷҗеҲ¶зһ¬ж—¶е№¶еҸ‘иҝһжҺҘж•°пјҢJavaзҡ„Semaphoreд№ҹеҸҜд»Ҙе®һзҺ°пјүгҖҒйҷҗеҲ¶ж—¶й—ҙзӘ—еҸЈеҶ…зҡ„е№іеқҮйҖҹзҺҮпјҲеҰӮGuavaзҡ„RateLimiterгҖҒnginxзҡ„limitreqжЁЎеқ—пјҢйҷҗеҲ¶жҜҸз§’зҡ„е№іеқҮйҖҹзҺҮпјүпјӣе…¶д»–иҝҳжңүеҰӮйҷҗеҲ¶иҝңзЁӢжҺҘеҸЈи°ғз”ЁйҖҹзҺҮгҖҒйҷҗеҲ¶MQзҡ„ж¶Ҳиҙ№йҖҹзҺҮгҖӮеҸҰеӨ–иҝҳеҸҜд»Ҙж №жҚ®зҪ‘з»ңиҝһжҺҘж•°гҖҒзҪ‘з»ңжөҒйҮҸгҖҒCPUжҲ–еҶ…еӯҳиҙҹиҪҪзӯүжқҘйҷҗжөҒгҖӮ

жҜ”еҰӮиҜҙпјҢжҲ‘д»¬йңҖиҰҒйҷҗеҲ¶ж–№жі•иў«и°ғз”Ёзҡ„е№¶еҸ‘ж•°дёҚиғҪи¶…иҝҮ100пјҲеҗҢдёҖж—¶й—ҙе№¶еҸ‘ж•°пјүпјҢеҲҷжҲ‘д»¬еҸҜд»Ҙз”ЁдҝЎеҸ·йҮҸ Semaphoreе®һзҺ°гҖӮеҸҜеҰӮжһңжҲ‘д»¬иҰҒйҷҗеҲ¶ж–№жі•еңЁдёҖж®өж—¶й—ҙеҶ…е№іеқҮиў«и°ғз”Ёж¬Ўж•°дёҚи¶…иҝҮ100пјҢеҲҷйңҖиҰҒдҪҝз”Ё RateLimiterгҖӮ

# йҷҗжөҒзҡ„еҹәзЎҖз®—жі•

жҲ‘д»¬е…ҲжқҘи®Іи§ЈдёҖдёӢдёӨдёӘйҷҗжөҒзӣёе…ізҡ„еҹәжң¬з®—жі•пјҡжјҸжЎ¶з®—жі•е’Ңд»ӨзүҢжЎ¶з®—жі•гҖӮ

д»ҺдёҠеӣҫдёӯпјҢжҲ‘д»¬еҸҜд»ҘзңӢеҲ°пјҢе°ұеғҸдёҖдёӘжјҸж–—дёҖж ·пјҢиҝӣжқҘзҡ„ж°ҙйҮҸе°ұеҘҪеғҸи®ҝй—®жөҒйҮҸдёҖж ·пјҢиҖҢеҮәеҺ»зҡ„ж°ҙйҮҸе°ұеғҸжҳҜжҲ‘д»¬зҡ„зі»з»ҹеӨ„зҗҶиҜ·жұӮдёҖж ·гҖӮеҪ“и®ҝй—®жөҒйҮҸиҝҮеӨ§ж—¶пјҢиҝҷдёӘжјҸж–—дёӯе°ұдјҡз§Ҝж°ҙпјҢеҰӮжһңж°ҙеӨӘеӨҡдәҶе°ұдјҡжәўеҮәгҖӮ

жјҸжЎ¶з®—жі•зҡ„е®һзҺ°еҫҖеҫҖдҫқиө–дәҺйҳҹеҲ—пјҢиҜ·жұӮеҲ°иҫҫеҰӮжһңйҳҹеҲ—жңӘж»ЎеҲҷзӣҙжҺҘж”ҫе…ҘйҳҹеҲ—пјҢз„¶еҗҺжңүдёҖдёӘеӨ„зҗҶеҷЁжҢүз…§еӣәе®ҡйў‘зҺҮд»ҺйҳҹеҲ—еӨҙеҸ–еҮәиҜ·жұӮиҝӣиЎҢеӨ„зҗҶгҖӮеҰӮжһңиҜ·жұӮйҮҸеӨ§пјҢеҲҷдјҡеҜјиҮҙйҳҹеҲ—ж»ЎпјҢйӮЈд№Ҳж–°жқҘзҡ„иҜ·жұӮе°ұдјҡиў«жҠӣејғгҖӮ

д»ӨзүҢжЎ¶з®—жі•еҲҷжҳҜдёҖдёӘеӯҳж”ҫеӣәе®ҡе®№йҮҸд»ӨзүҢзҡ„жЎ¶пјҢжҢүз…§еӣәе®ҡйҖҹзҺҮеҫҖжЎ¶йҮҢж·»еҠ д»ӨзүҢгҖӮжЎ¶дёӯеӯҳж”ҫзҡ„д»ӨзүҢж•°жңүжңҖеӨ§дёҠйҷҗпјҢи¶…еҮәд№ӢеҗҺе°ұиў«дёўејғжҲ–иҖ…жӢ’з»қгҖӮеҪ“жөҒйҮҸжҲ–иҖ…зҪ‘з»ңиҜ·жұӮеҲ°иҫҫж—¶пјҢжҜҸдёӘиҜ·жұӮйғҪиҰҒиҺ·еҸ–дёҖдёӘд»ӨзүҢпјҢеҰӮжһңиғҪеӨҹиҺ·еҸ–еҲ°пјҢеҲҷзӣҙжҺҘеӨ„зҗҶпјҢе№¶дё”д»ӨзүҢжЎ¶еҲ йҷӨдёҖдёӘд»ӨзүҢгҖӮеҰӮжһңиҺ·еҸ–дёҚеҗҢпјҢиҜҘиҜ·жұӮе°ұиҰҒиў«йҷҗжөҒпјҢиҰҒд№ҲзӣҙжҺҘдёўејғпјҢиҰҒд№ҲеңЁзј“еҶІеҢәзӯүеҫ…гҖӮ

д»ӨзүҢжЎ¶е’ҢжјҸжЎ¶еҜ№жҜ”пјҡ

  • д»ӨзүҢжЎ¶жҳҜжҢүз…§еӣәе®ҡйҖҹзҺҮеҫҖжЎ¶дёӯж·»еҠ д»ӨзүҢпјҢиҜ·жұӮжҳҜеҗҰиў«еӨ„зҗҶйңҖиҰҒзңӢжЎ¶дёӯд»ӨзүҢжҳҜеҗҰи¶іеӨҹпјҢеҪ“д»ӨзүҢж•°еҮҸдёәйӣ¶ж—¶еҲҷжӢ’з»қж–°зҡ„иҜ·жұӮпјӣжјҸжЎ¶еҲҷжҳҜжҢүз…§еёёйҮҸеӣәе®ҡйҖҹзҺҮжөҒеҮәиҜ·жұӮпјҢжөҒе…ҘиҜ·жұӮйҖҹзҺҮд»»ж„ҸпјҢеҪ“жөҒе…Ҙзҡ„иҜ·жұӮж•°зҙҜз§ҜеҲ°жјҸжЎ¶е®№йҮҸж—¶пјҢеҲҷж–°жөҒе…Ҙзҡ„иҜ·жұӮиў«жӢ’з»қпјӣ
  • д»ӨзүҢжЎ¶йҷҗеҲ¶зҡ„жҳҜе№іеқҮжөҒе…ҘйҖҹзҺҮпјҢе…Ғи®ёзӘҒеҸ‘иҜ·жұӮпјҢеҸӘиҰҒжңүд»ӨзүҢе°ұеҸҜд»ҘеӨ„зҗҶпјҢж”ҜжҢҒдёҖж¬ЎжӢҝ3дёӘд»ӨзүҢпјҢ4дёӘд»ӨзүҢпјӣжјҸжЎ¶йҷҗеҲ¶зҡ„жҳҜеёёйҮҸжөҒеҮәйҖҹзҺҮпјҢеҚіжөҒеҮәйҖҹзҺҮжҳҜдёҖдёӘеӣәе®ҡеёёйҮҸеҖјпјҢжҜ”еҰӮйғҪжҳҜ1зҡ„йҖҹзҺҮжөҒеҮәпјҢиҖҢдёҚиғҪдёҖж¬ЎжҳҜ1пјҢдёӢж¬ЎеҸҲжҳҜ2пјҢд»ҺиҖҢе№іж»‘зӘҒеҸ‘жөҒе…ҘйҖҹзҺҮпјӣ
  • д»ӨзүҢжЎ¶е…Ғи®ёдёҖе®ҡзЁӢеәҰзҡ„зӘҒеҸ‘пјҢиҖҢжјҸжЎ¶дё»иҰҒзӣ®зҡ„жҳҜе№іж»‘жөҒеҮәйҖҹзҺҮпјӣ

# Guava RateLimiter

GuavaжҳҜJavaйўҶеҹҹдјҳз§Җзҡ„ејҖжәҗйЎ№зӣ®пјҢе®ғеҢ…еҗ«дәҶGoogleеңЁJavaйЎ№зӣ®дёӯдҪҝз”ЁдёҖдәӣж ёеҝғеә“пјҢеҢ…еҗ«йӣҶеҗҲ(Collections)пјҢзј“еӯҳ(Caching)пјҢе№¶еҸ‘зј–зЁӢеә“(Concurrency)пјҢеёёз”ЁжіЁи§Ј(Common annotations)пјҢStringж“ҚдҪңпјҢI/Oж“ҚдҪңж–№йқўзҡ„дј—еӨҡйқһеёёе®һз”Ёзҡ„еҮҪж•°гҖӮ Guavaзҡ„ RateLimiterжҸҗдҫӣдәҶд»ӨзүҢжЎ¶з®—жі•е®һзҺ°пјҡе№іж»‘зӘҒеҸ‘йҷҗжөҒ(SmoothBursty)е’Ңе№іж»‘йў„зғӯйҷҗжөҒ(SmoothWarmingUp)е®һзҺ°гҖӮ

RateLimiterзҡ„зұ»еӣҫеҰӮдёҠжүҖзӨәпјҢе…¶дёӯ RateLimiterжҳҜе…ҘеҸЈзұ»пјҢе®ғжҸҗдҫӣдәҶдёӨеҘ—е·ҘеҺӮж–№жі•жқҘеҲӣе»әеҮәдёӨдёӘеӯҗзұ»гҖӮиҝҷеҫҲз¬ҰеҗҲгҖҠEffective JavaгҖӢдёӯзҡ„з”ЁйқҷжҖҒе·ҘеҺӮж–№жі•д»Јжӣҝжһ„йҖ еҮҪж•°зҡ„е»әи®®пјҢжҜ•з«ҹиҜҘд№Ұзҡ„дҪңиҖ…д№ҹжӯЈжҳҜGuavaеә“зҡ„дё»иҰҒз»ҙжҠӨиҖ…пјҢдәҢиҖ…й…ҚеҗҲ"йЈҹз”Ё"жӣҙдҪігҖӮ

// RateLimiterжҸҗдҫӣдәҶдёӨдёӘе·ҘеҺӮж–№жі•пјҢжңҖз»Ҳдјҡи°ғз”ЁдёӢйқўдёӨдёӘеҮҪж•°пјҢз”ҹжҲҗRateLimiterзҡ„дёӨдёӘеӯҗзұ»гҖӮ
static RateLimiter create(SleepingStopwatch stopwatch, double permitsPerSecond) {
 RateLimiter rateLimiter = new SmoothBursty(stopwatch, 1.0 /* maxBurstSeconds */);
    rateLimiter.setRate(permitsPerSecond);
 return rateLimiter;
}
static RateLimiter create(
 SleepingStopwatch stopwatch, double permitsPerSecond, long warmupPeriod, TimeUnit unit,
 double coldFactor) {
 RateLimiter rateLimiter = new SmoothWarmingUp(stopwatch, warmupPeriod, unit, coldFactor);
    rateLimiter.setRate(permitsPerSecond);
 return rateLimiter;
}

зҙ§жҺҘзқҖпјҢзәҝзЁӢдәҢд№ҹжқҘacquire(2)пјҢйҰ–е…Ҳ resyncеҮҪж•°еҸ‘зҺ°еҪ“еүҚж—¶й—ҙж—©дәҺ nextFreeTicketMicrosпјҢжүҖд»Ҙж— жі•еўһеҠ д»ӨзүҢж•°пјҢжүҖд»ҘйңҖиҰҒйў„ж”Ҝд»ҳ2дёӘд»ӨзүҢпјҢnextFreeTicketMicros= nextFreeTicketMicro + 2 * 500 = 155391849 7748гҖӮзәҝзЁӢдәҢйңҖиҰҒзӯүеҫ…155391849 6748ж—¶еҲ»пјҢд№ҹе°ұжҳҜзәҝзЁӢдёҖиҺ·еҸ–ж—¶и®Ўз®—зҡ„nextFreeTicketMicrosж—¶еҲ»гҖӮеҗҢж ·зҡ„пјҢзәҝзЁӢдёүиҺ·еҸ–д»ӨзүҢж—¶д№ҹйңҖиҰҒзӯүеҫ…еҲ°зәҝзЁӢдәҢи®Ўз®—зҡ„nextFreeTicketMicrosж—¶еҲ»гҖӮ

# е№іж»‘йў„зғӯйҷҗжөҒ

дёҠиҝ°е°ұжҳҜе№іж»‘зӘҒеҸ‘йҷҗжөҒRateLimiterзҡ„е®һзҺ°пјҢдёӢйқўжҲ‘д»¬жқҘзңӢдёҖдёӢеҠ дёҠйў„зғӯзј“еҶІжңҹзҡ„е®һзҺ°еҺҹзҗҶгҖӮ SmoothWarmingUpе®һзҺ°йў„зғӯзј“еҶІзҡ„е…ій”®еңЁдәҺе…¶еҲҶеҸ‘д»ӨзүҢзҡ„йҖҹзҺҮдјҡйҡҸж—¶й—ҙе’Ңд»ӨзүҢж•°иҖҢж”№еҸҳпјҢйҖҹзҺҮдјҡе…Ҳж…ўеҗҺеҝ«гҖӮиЎЁзҺ°еҪўејҸеҰӮдёӢеӣҫжүҖзӨәпјҢд»ӨзүҢеҲ·ж–°зҡ„ж—¶й—ҙй—ҙйҡ”з”ұй•ҝйҖҗжёҗеҸҳзҹӯгҖӮзӯүеӯҳеӮЁд»ӨзүҢж•°д»ҺmaxPermitsеҲ°иҫҫthresholdPermitsж—¶пјҢеҸ‘ж”ҫд»ӨзүҢзҡ„ж—¶й—ҙд»·ж јд№ҹз”ұcoldIntervalйҷҚдҪҺеҲ°дәҶжӯЈеёёзҡ„stableIntervalгҖӮ

SmoothWarmingUpзҡ„зӣёе…ід»Јз ҒеҰӮдёӢжүҖзӨәпјҢзӣёе…ізҡ„йҖ»иҫ‘йғҪеҶҷеңЁжіЁйҮҠдёӯгҖӮ

// SmoothWarmingUpпјҢзӯүеҫ…ж—¶й—ҙе°ұжҳҜи®Ўз®—дёҠеӣҫдёӯжўҜеҪўжҲ–иҖ…жӯЈж–№еҪўзҡ„йқўз§ҜгҖӮ
long storedPermitsToWaitTime(double storedPermits, double permitsToTake) {
 /**
    * еҪ“еүҚpermitsи¶…еҮәйҳҲеҖјзҡ„йғЁеҲҶ
    */
 double availablePermitsAboveThreshold = storedPermits - thresholdPermits;
 long micros = 0;
 /**
    * еҰӮжһңеҪ“еүҚеӯҳеӮЁзҡ„д»ӨзүҢж•°и¶…еҮәthresholdPermits
    */
 if (availablePermitsAboveThreshold > 0.0) {
 /**
     * еңЁйҳҲеҖјеҸідҫ§е№¶дё”йңҖиҰҒиў«ж¶ҲиҖ—зҡ„д»ӨзүҢж•°йҮҸ
     */
 double permitsAboveThresholdToTake = min(availablePermitsAboveThreshold, permitsToTake);
 
 /**
        * жўҜеҪўзҡ„йқўз§Ҝ
        *
        * й«ҳ * (йЎ¶ * еә•) / 2
        *
        * й«ҳжҳҜ permitsAboveThresholdToTake д№ҹе°ұжҳҜеҸідҫ§йңҖиҰҒж¶Ҳиҙ№зҡ„д»ӨзүҢж•°
        * еә• иҫғй•ҝ permitsToTime(availablePermitsAboveThreshold)
        * йЎ¶ иҫғзҹӯ permitsToTime(availablePermitsAboveThreshold - permitsAboveThresholdToTake)
        */
    micros = (long) (permitsAboveThresholdToTake
 * (permitsToTime(availablePermitsAboveThreshold)
 + permitsToTime(availablePermitsAboveThreshold - permitsAboveThresholdToTake)) / 2.0);
 /**
        * еҮҸеҺ»е·Із»ҸиҺ·еҸ–зҡ„еңЁйҳҲеҖјеҸідҫ§зҡ„д»ӨзүҢж•°
        */
    permitsToTake -= permitsAboveThresholdToTake;
 }
 /**
    * е№ізЁіж—¶жңҹзҡ„йқўз§ҜпјҢжӯЈеҘҪжҳҜй•ҝд№ҳе®Ҫ
    */
    micros += (stableIntervalMicros * permitsToTake);
 return micros;
}
 
double coolDownIntervalMicros() {
 /**
    * жҜҸз§’еўһеҠ зҡ„д»ӨзүҢж•°дёә warmupж—¶й—ҙ/maxPermits. иҝҷж ·зҡ„иҜқпјҢеңЁwarmuptimeж—¶й—ҙеҶ…пјҢе°ұе°ұеўһеј зҡ„д»ӨзүҢж•°йҮҸ
    * дёә maxPermits
    */
 return warmupPeriodMicros / maxPermits;
}
